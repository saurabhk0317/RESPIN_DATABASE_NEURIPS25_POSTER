    % ===== COLUMN 1 =====
    \begin{PosterBlock}{1. Motivation}
    \large
    \justifying
    India has hundreds of dialects. Standard ASR models often
    collapse accent / dialect variation, causing:
    \begin{itemize}
        \item High WER for minority dialects
        \item Misrouting in voice assistants / call centers
        \item Unfair downstream analytics
    \end{itemize}

    Our goal:
    \begin{itemize}
        \item Build ASR that \textbf{knows which dialect it is hearing}
        \item Improve both \textbf{transcription quality} and \textbf{dialect ID} jointly
    \end{itemize}
    \end{PosterBlock}

    \begin{PosterBlock}{2. Key Idea (High-level)}
    \large
    \justifying
    \textbf{We fuse acoustic + ASR decoder + text embedding features}
    into a unified dialect representation:
    \begin{itemize}
        \item Conformer / SSL-ASR encoder captures speech acoustics
        \item Decoder hypotheses give transcript-aware signals
        \item Lightweight RoBERTa-style text encoder refines dialect cues
    \end{itemize}

    Training is \textbf{multi-task}: ASR loss + dialect loss.
    \end{PosterBlock}

    \begin{PosterBlock}{3. Contributions}
    \large
    \begin{itemize}
        \item A joint ASR+DID architecture that uses ASR internals as dialect clues
        \item Results on 9 Indian languages, 30+ dialects
        \item Consistent gains on minority dialects with modest GPU budget
    \end{itemize}
    \end{PosterBlock}

\end{minipage}%

    \begin{PosterBlock}{4. Method / Architecture}
    \large
    \justifying
    \textbf{Pipeline overview:}
    \begin{enumerate}
        \item SSL-initialized Conformer encoder produces frame-level speech reps
        \item Hybrid CTC/Attention decoder outputs transcript
        \item Hypothesized transcript is fed into a compact RoBERTa-like text encoder
        \item We fuse (a) acoustic embedding, (b) decoder state, (c) text embedding
        \item A shared classifier predicts dialect ID
    \end{enumerate}
